2 Combined Visualizations

earthquake %>%
  filter(State %in% top_10_states) %>%
  ggplot(aes(x = Distance_Category, y = Magnitude, color = State)) +

  facet_wrap(~State) +
  labs(
    title = "Magnitude vs Distance by State",

4 Heatmap Visualization

4.1 Intensity Heatmap by State and Distance

earthquake %>%
  filter(State %in% top_10_states) %>%
  filter(!is.na(Intensity)) %>%
  group_by(State, Distance_Category) %>%
  summarise(Mean_Intensity = mean(Intensity), .groups = "drop") %>%
  ggplot(aes(x = Distance_Category, y = State, fill = Mean_Intensity)) +
  geom_tile(color = "white") +
  geom_text(aes(label = round(Mean_Intensity, 1)), color = "white", size = 5) +
  scale_fill_gradient(low = "blue", high = "red") +
  labs(
    title = "Mean Intensity by State and Distance",
    x = "Distance Category",
    y = "State",
    fill = "Mean Intensity"
  ) +
  theme_minimal()

6.1 Location Popularity and Distance

earthquake %>%
  group_by(Location, State) %>%
  summarise(
    Count = n(),
    Mean_Distance = mean(Distance_km, na.rm = TRUE),
    Mean_Intensity = mean(Intensity, na.rm = TRUE),
    .groups = "drop"
  ) %>%
  filter(Count > 5) %>%
  ggplot(aes(x = Mean_Distance, y = Count, size = Mean_Intensity, color = State)) +

7 Bar Chart with Error Bars

7.1 Mean Intensity by Distance Category (with Standard Error)

earthquake %>%
  filter(!is.na(Intensity)) %>%
  group_by(Distance_Category) %>%
  summarise(
    Mean = mean(Intensity),
    SE = sd(Intensity) / sqrt(n()),
    .groups = "drop"
  ) %>%
  ggplot(aes(x = Distance_Category, y = Mean, fill = Distance_Category)) +
  geom_bar(stat = "identity", alpha = 0.7) +
  geom_errorbar(aes(ymin = Mean - SE, ymax = Mean + SE), width = 0.2) +
  labs(
    title = "Mean Intensity by Distance (with Standard Error)",
    x = "Distance Category",
    y = "Mean Intensity"
  ) +
  theme_minimal() +
  theme(legend.position = "none")
