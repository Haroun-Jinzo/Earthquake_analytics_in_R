1.2 Load Data

earthquake <- readRDS("../data/processed/earthquake_clean.rds")
cat("Loaded", nrow(earthquake), "observations\n")
Loaded 142083 observations
top_10_states <- earthquake %>%
  count(State, sort = TRUE) %>%
  head(10) %>%
  pull(State)

2 Magnitude Analysis

2.1 Histogram of Magnitudes

ggplot(earthquake, aes(x = Magnitude)) +
  geom_histogram(binwidth = 0.5, fill = "steelblue", color = "black") +
  labs(
    title = "Distribution of Earthquake Magnitudes",
    x = "Magnitude",
    y = "Count"
  ) +
  theme_minimal()

2.2 Magnitude by State

earthquake %>%
  filter(State %in% top_10_states) %>%
  ggplot(aes(x = State, y = Magnitude, fill = State)) +
  geom_boxplot() +
  labs(
    title = "Magnitude by State (Top 10 States)",
    x = "State",
    y = "Magnitude"
  ) +
  theme_minimal() +
  theme(legend.position = "none")

3 Geographic Analysis

3.1 Where are the Epicenters?

ggplot(earthquake, aes(x = Epicenter_Lon, y = Epicenter_Lat)) +
  geom_point(aes(size = Magnitude, color = Magnitude), alpha = 0.6) +
  scale_color_gradient(low = "yellow", high = "red") +
  labs(
    title = "Earthquake Epicenter Locations",
    x = "Longitude",
    y = "Latitude"
  ) +
  theme_minimal()
Warning: Removed 6463 rows containing missing values or values outside the scale range
(`geom_point()`).

Observation: Epicenters are concentrated around 39.5°N, -120°W (Lake Tahoe area).

6 Summary

Key findings from exploratory analysis:

  • Most earthquakes were magnitude 8
  • Epicenter located near Lake Tahoe (CA/NV border)
  • Recording stations up to 300 km away
  • Activity concentrated over 2 days
